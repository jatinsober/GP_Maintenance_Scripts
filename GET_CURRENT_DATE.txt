SET SEARCH_PATH TO RISK;
CREATE OR REPLACE FUNCTION GET_CURRENT_DATE(CURR_DATE DATE) RETURNS
 DATE AS $$
DECLARE
	SYS_DATE DATE;   
BEGIN

	 /*GET SYSTEM DATE*/
	 SELECT CURRENT_DATE INTO SYS_DATE;
	 RETURN SYS_DATE;
	 
END;
$$
LANGUAGE PLPGSQL;

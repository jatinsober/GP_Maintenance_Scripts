SET SEARCH_PATH TO RISK;
CREATE OR REPLACE VIEW ADM_SOURCE_INFO_V
(SOURCE_FILE, EXTERNAL_TABLE, BUSINESS_DATE, FIELD, APPL_ID, 
 SOURCE_FILE_DESC, SOURCE_FILE_STATUS, PROCESS_STATUS)
AS 
SELECT B.ASI_FILE, B.ASI_EXT_TABLE, B.ASI_BUS_DATE, B.ASI_BUS_DATE_FIELD,  
B.ASI_APPL_ID, B.ASI_COMMENT, B.ASI_FILE_STATUS, B.ASI_PRO_STATUS  
FROM ADM_SOURCE_INFO B;

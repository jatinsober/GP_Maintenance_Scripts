SET SEARCH_PATH TO RISK;
CREATE OR REPLACE VIEW ADM_PROCESS_LOG_V
(BUSINESS_DATE, PROCESS_NAME, PROCESS_STATUS, LOADING_STATUS, SYSTEM_DATE)
AS 
SELECT A.APL_PRO_BUS_DATE, A.APL_PRO_NAME, A.APL_PRO_STATUS, A.APL_LOAD_STATUS, A.APL_SYSTEM_DATE 
FROM ADM_PROCESS_LOG A 
ORDER BY A.APL_SYSTEM_DATE;
